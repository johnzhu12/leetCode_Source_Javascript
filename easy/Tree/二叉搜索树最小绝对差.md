# [二叉搜索树最小绝对差](https://leetcode-cn.com/problems/minimum-absolute-difference-in-bst/)

```js
var getMinimumDifference = function (root) {
  let ans = Number.MAX_SAFE_INTEGER,
    pre = -1
  const dfs = (root) => {
    if (root === null) {
      return
    }
    dfs(root.left)
    if (pre == -1) {
      pre = root.val
    } else {
      ans = Math.min(ans, root.val - pre)
      pre = root.val
    }
    dfs(root.right)
  }
  dfs(root)
  return ans
}
```
